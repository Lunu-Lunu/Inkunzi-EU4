# For Government Reforms to identify properly influence and loyalty modifiers, remember to put disqualifying triggers before the has_reform

estate_nimbus = {
	icon = 3

	# If true, country will get estate
	trigger = {
		OR = {
			has_country_flag = primary_nimbus_nation
			has_country_flag = nimbus_accepted
		}
		NOT = { has_government_attribute = is_merchant_republic }
		NOT = { has_reform = trading_city }
		NOT = { is_revolutionary_republic_trigger = yes }
		NOT = { government = native }
		NOT = { has_reform = cossacks_reform }
		NOT = { has_reform = siberian_tribe }
		NOT = { has_reform = veche_republic }
		NOT = { has_reform = celestial_empire }
		NOT = { has_reform = pirate_republic_reform }
		is_nomad = no
		NOT = { has_reform = revolutionary_empire_reform }
	}

	# These scale with loyalty & power
	country_modifier_happy = {
		diplomatic_reputation = 2
		diplomatic_annexation_cost = -0.2
		vassal_income = 0.25
	}
	country_modifier_neutral = {
		diplomatic_reputation = 2
		vassal_income = 0.25
	}	
	country_modifier_angry = {
		diplomatic_reputation = -2
		diplomatic_annexation_cost = 0.2
		vassal_income = -0.5
		global_unrest = 2
	}
	land_ownership_modifier = {
		nimbus_loyalty_modifier = 0.2
	}
	
	province_independence_weight = {
		factor = 1
		modifier = {
			factor = 0
			NOT = { culture = nimbus }
		}
		modifier = {
			factor = 2
			has_owner_culture = no
		}
		modifier = {
			factor = 0.67
			NOT = { is_state_core = owner }
		}
	}

	# Influence modifiers
	base_influence = 10
	influence_modifier = {
		desc = ESTATE_NIMBUS_HAS_SUBJECT
		trigger = {
			num_of_subjects = 1
		}
		influence = 15	
	}
	influence_modifier = {
		desc = ESTATE_NIMBUS_PROMOTED
		trigger = {
			has_country_flag = nimbus_accepted
			NOT = { has_country_flag = primary_nimbus_nation }
		}
		influence = 5
	}
	influence_modifier = {
		desc = ESTATE_HUMANS_PROMOTED
		trigger = {
			has_country_flag = humans_accepted
		}
		influence = 10
	}
	influence_modifier = {
		desc = EST_VAL_OTHER_ESTATE_IN_POWER
		trigger = {
			OR = {
				has_disaster = estate_nobility_disaster
				has_disaster = estate_church_disaster
				has_disaster = estate_burghers_disaster
				has_disaster = estate_brahmins_disaster
				has_disaster = estate_vaisyas_disaster
				has_disaster = estate_jains_disaster
			}
		}
		influence = -40
	}
	
	#Loyalty Modifiers
	loyalty_modifier = {
		desc = EST_VAL_OTHER_ESTATE_IN_POWER_LOY
		trigger = {
			OR = {
				has_disaster = estate_nobility_disaster
				has_disaster = estate_church_disaster
				has_disaster = estate_burghers_disaster
				has_disaster = estate_brahmins_disaster
				has_disaster = estate_vaisyas_disaster
				has_disaster = estate_jains_disaster
			}
		}
		loyalty = -20
	}
	#Rajasthani Culture Promoted
	loyalty_modifier = {
		desc = ESTATE_NIMBUS_IS_PRIMARY
		trigger = {
			has_country_flag = primary_nimbus_nation
		}
		loyalty = 5
	}
	loyalty_modifier = {
		desc = ESTATE_NIMBUS_HAS_SUBJECT_LOY
		trigger = {
			num_of_subjects = 1
		}
		loyalty = 5
	}

	privileges = {
		estate_nimbus_land_rights
	}

	agendas = {
		estate_nimbus_vassalise_vulnerable_country
	}

	color = { 200 200 255 }
	
	influence_from_dev_modifier = 1.0
}
